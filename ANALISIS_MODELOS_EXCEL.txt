CREATE TABLE articulos (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  codigo TEXT UNIQUE,
  nombre TEXT NOT NULL,
  unidad TEXT,
  coste_kilo REAL,
  tipo TEXT, -- 'Ingrediente', 'Producto'
  proveedor_id INTEGER,
  grupo_conservacion TEXT,
  activo BOOLEAN DEFAULT 1,
  created_at DATETIME,
  updated_at DATETIME
);

CREATE TABLE escandallos (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  codigo_plato TEXT,
  articulo_id INTEGER,
  cantidad REAL,
  unidad TEXT,
  coste_total REAL,
  FOREIGN KEY (articulo_id) REFERENCES articulos(id)
);

CREATE TABLE partidas_cocina (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  nombre TEXT NOT NULL,
  responsable TEXT,
  descripcion TEXT,
  activo BOOLEAN DEFAULT 1
);

CREATE TABLE trazabilidad_platos (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  plato_id INTEGER,
  fecha_produccion DATE,
  dia_semana TEXT,
  anticipado BOOLEAN,
  estado TEXT,
  FOREIGN KEY (plato_id) REFERENCES platos(id)
);

CREATE TABLE control_sanidad (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  plato_id INTEGER,
  articulo_id INTEGER,
  fecha_produccion DATE,
  punto_critico TEXT,
  punto_corrector TEXT,
  FOREIGN KEY (plato_id) REFERENCES platos(id),
  FOREIGN KEY (articulo_id) REFERENCES articulos(id)
);

CREATE TABLE inventario (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  articulo_id INTEGER,
  cantidad REAL,
  fecha_registro DATE,
  FOREIGN KEY (articulo_id) REFERENCES articulos(id)
);

CREATE TABLE etiquetas (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  nombre TEXT,
  numero_etiquetas INTEGER,
  info_trazabilidad TEXT,
  anticipado BOOLEAN
);
